language: cpp
dist: xenial
sudo: required
git:
  depth: 3
env:
  matrix:
  PACKAGES_MATRIX

compiler: clang
install:
  - echo "$PWD"
  - if [ -n "$TRAVIS_PULL_REQUEST" ] && [ "$PACKAGE" != CHECK ]; then  DO_IGNORE=FALSE;  for ARG in $(echo "$PACKAGE");do if [ "$ARG" = "Maintenance" ]; then continue; fi; . $PWD/.travis/test_package.sh "$PWD" "$ARG";  echo "DO_IGNORE is $DO_IGNORE";  if [ "$DO_IGNORE" = "FALSE" ]; then  break; fi; done; if [ "$DO_IGNORE" = "TRUE" ]; then travis_terminate 0; fi;fi
  - /usr/bin/time  bash .travis/install.sh
  - export CXX=clang++ CC=clang;
before_script:
  - mkdir -p build
  - cd build
  - /usr/bin/time  cmake -DCMAKE_CXX_FLAGS="-std=c++11" -DCGAL_HEADER_ONLY=ON -DCMAKE_CXX_FLAGS_RELEASE=-DCGAL_NDEBUG -DWITH_examples=ON -DWITH_demos=ON -DWITH_tests=ON ..
  - /usr/bin/time  make
  - /usr/bin/time  sudo make install &>/dev/null
  - cd ..
script:
  - cd ./.travis
  - /usr/bin/time  bash ./build_package.sh $PACKAGE
notifications:
  email:
    on_success: change
    # default: always
    on_failure: always
    # default: always
